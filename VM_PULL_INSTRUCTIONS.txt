MANUAL SETUP (Step-by-step)
═════════════════════════════════════════════════════════════════════════════

# Step 1: Update system
sudo apt-get update
sudo apt-get upgrade -y

# Step 2: Install dependencies
sudo apt-get install -y \
    clang llvm libbpf-dev \
    linux-headers-$(uname -r) \
    golang-go coq make git \
    netcat-openbsd python3

# Step 3: CRITICAL - Fix Ubuntu 25.10 asm symlink
# This is ESSENTIAL for eBPF compilation
sudo ln -sf /usr/include/x86_64-linux-gnu/asm /usr/include/asm

# Verify it worked
ls -la /usr/include/asm
# Should show: asm -> /usr/include/x86_64-linux-gnu/asm

# Step 4: Clone repository (first time)
cd $HOME
git clone https://github.com/dyb5784/dwell-fiber.git
cd dwell-fiber

# OR: Update existing repo
cd dwell-fiber
git fetch origin
git pull origin main

# Step 5: Build everything
make clean all

# This builds:
#   ✓ BPF programs (bpf/dwell_monitor.bpf.o)
#   ✓ Coq proofs (coq/dwell_stable.vo)
#   ✓ Go daemon (bin/dwell-fiber-daemon)

# Step 6: Verify build
ls -la bin/dwell-fiber-daemon
file bin/dwell-fiber-daemon  # Should show: ELF 64-bit

TROUBLESHOOTING

  $ cd test && go run workload_generator.go

  $ curl http://localhost:9090/metrics
  $ ps aux | grep dwell-fiber
